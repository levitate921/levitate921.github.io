    A. Alice and Bob¶

    发现答案肯定是 \(a+1,a-1\) 其一，计算数组中是 \(>a\) 的多还是 \(<a\) 的多即可。

    code 
    // Problem: A. Alice and Bob
// Contest: Codeforces - Educational Codeforces Round 184 (Rated for Div. 2)
// URL: https://codeforces.com/contest/2169/problem/A
// Memory Limit: 512 MB
// Time Limit: 2000 ms
// 
// Powered by CP Editor (https://cpeditor.org)

#include <bits/stdc++.h>
using namespace std;

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int T, n, a;
    cin >> T;
    while(T--){
        cin >> n >> a;
        vector<int> v(n);
        int le = 0, ri = 0;
        for(int i = 0; i < n; ++ i){
            int x;
            cin >> x;
            v.push_back(x);
            if(x < a){
                ++ le;
            } else if(x > a){
                ++ ri;
            }
        }
        if(le <= ri){
            cout << a + 1 << '\n'; 
        } else {
            cout << a - 1 << '\n';
        }
    }
    return 0;
}

    // Problem: B. Drifting Away
// Contest: Codeforces - Educational Codeforces Round 184 (Rated for Div. 2)
// URL: https://codeforces.com/contest/2169/problem/B
// Memory Limit: 512 MB
// Time Limit: 2000 ms
// 
// Powered by CP Editor (https://cpeditor.org)

#include <bits/stdc++.h>
using namespace std;

int main(){
    int T;
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin >> T;
    while(T--){
        string s;
        cin >> s;
        if(s.size() == 1){
            cout << "1\n";
        } else {
            int flg = 0;
            for(int i = 1; i < s.size(); ++ i){
                if(s[i-1] != '<' && s[i] == '<'){
                    flg = 1;
                }
                if(s[i-1] == '>' && s[i] != '>'){
                    flg = 1;
                }
                if(s[i-1] == '*' && s[i] == '*'){
                    flg = 1;
                }
            }
            if(flg){
                cout << "-1\n";
            } else {
                int l = -1, r = s.size();
                for(int i = 0; i < s.size(); ++ i){
                    if(s[i] == '<'){
                        l = max(l, i);
                    }
                    if(s[i] == '>'){
                        r = min(r, i);
                    }
                }
                cout << max((int)s.size()-1-l, r) << '\n';
            }
        }
    }
    return 0;
}

    C. Range Operation¶

    拆式子：\(sum_r-sum_{l-1} + (l+r)(r-l+1)\)，会发现根本没有 \(lr\) 这一项，那么就从左往右扫一下记录最优的 \(l\)，然后每次把目前的 \(i\) 当作 \(r\) 更新一下答案就行。

    code 
    // Problem: C. Range Operation
// Contest: Codeforces - Educational Codeforces Round 184 (Rated for Div. 2)
// URL: https://codeforces.com/contest/2169/problem/C
// Memory Limit: 512 MB
// Time Limit: 2000 ms
// 
// Powered by CP Editor (https://cpeditor.org)

#include <bits/stdc++.h>
using namespace std;

/*
(r+l) * (r-l+1) - sum[r] + sum[l-1]
r^2+r-l^2+l-sum[r]+sum[l-1]
*/

int main(){
    int T, n;
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin >> T;
    while(T--){
        cin >> n;
        vector<int> a(n);
        vector<long long> sum(n);
        long long mx = 0, ans = -1e18;
        for(int i = 0; i < n; ++ i){
            cin >> a[i];
            sum[i] = a[i];
            if(i){
                sum[i] += sum[i-1];
                mx = max(mx, - 1ll * (i+1) * (i+1) + i+1 + sum[i-1]);
            }
            ans = max(ans, mx + 1ll * (i+1) * (i+1) + (i+1) - sum[i]);
        }
        cout << ans + sum[n-1] << '\n';
    }
    return 0;
}

    D1/D2. Removal of a Sequence¶

    倒着做，设目前第 \(k\) 位为 \(k\)，然后每次 \(k\leftarrow k + \lfloor\dfrac{k-1}{y-1}\rfloor\)，最后判断 \(k\) 是否 \(\leq 10^{12}\) 即可。直接模拟通过 D1。

    把形式写的好看一点：\(k,y\) 都减一，然后每次变为 \(k\leftarrow k + \lfloor \dfrac ky\rfloor\)。

    考虑把每次 \(\lfloor \dfrac ky\rfloor\) 为固定值的段提取出来一次计算。

    分析复杂度：当 \(\lfloor \dfrac ky\rfloor \leq 10^6\) 的时候，只会计算至多 \(10^6\) 次；否则，每次变化 \(k\) 至少加 \(10^6\)，总共至多加 \(10^6\) 次，所以复杂度是 \(O(T\log V)\) 的。

    code 
    // Problem: D1. Removal of a Sequence (Easy Version)
// Contest: Codeforces - Educational Codeforces Round 184 (Rated for Div. 2)
// URL: https://codeforces.com/contest/2169/problem/D1
// Memory Limit: 512 MB
// Time Limit: 2000 ms
// 
// Powered by CP Editor (https://cpeditor.org)

#include <bits/stdc++.h>
using namespace std;

int main(){
    int T;
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin >> T;
    long long n = 1e12;
    while(T--){
        long long x, y, k;
        cin >> x >> y >> k;
        if(y == 1){
            cout << "-1\n";
        } else {
            while(x --){
                k += (k - 1) / (y - 1);
                if(k > n){
                    cout << "-1\n";
                    break;
                }
            }
            if(k <= n){
                cout << k << '\n';
            }
        }
    }
    return 0;
}

    E. Points Selection¶

    转化一下题意，就能知道是求从点集中选择一些点，使得最小的包含所选点的矩形的周长 + 未选点点权和最大。

    那么发现所选点一定都在这个矩形的边上，可能在边上，可能在角上，那么就对于每种情况（比如，四个点都在边上，或者左上角一个点，右下两条边上两个点之类的情况）都拆贡献算就行。

    code 
    //CF2169E
#include <bits/stdc++.h>
using namespace std;

int main(){
    int T, n;
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin >> T;
    while(T--){
        cin >> n;
        vector<long long> x(n), y(n), c(n);
        for(int i = 0; i < n; ++ i){
            cin >> x[i];
        }
        for(int i = 0; i < n; ++ i){
            cin >> y[i];
        }
        long long sum = 0;
        for(int i = 0; i < n; ++ i){
            cin >> c[i];
            sum += c[i];
        }
        long long U, D, L, R, UL, UR, DL, DR, UDLR;
        U = D = L = R = UL = UR = DL = DR = UDLR = -1e17;
        for(int i = 0; i < n; ++ i){
            U = max(U, 2 * x[i] - c[i]);
            D = max(D, - 2 * x[i] - c[i]);
            L = max(L, - 2 * y[i] - c[i]);
            R = max(R, 2 * y[i] - c[i]);
            UL = max(UL, 2 * x[i] - 2 * y[i] - c[i]);
            UR = max(UR, 2 * x[i] + 2 * y[i] - c[i]);
            DL = max(DL, - 2 * x[i] - 2 * y[i] - c[i]);
            DR = max(DR, - 2 * x[i] + 2 * y[i] - c[i]);
            UDLR = max(UDLR, - c[i]);
        }
        long long ans = UDLR;
        ans = max(ans, max(UL, U + L) + max(DR, D + R));
        ans = max(ans, max(UR, U + R) + max(DL, D + L));
        cout << ans + sum << '\n';
    }
    return 0;
}

    F. Subsequence Problem¶

    题意

    你有 \(q\) 个长度 \(\leq 5\) 的序列，我们定义一个长度为 \(q\) 的序列 \(a\) 可以被这 \(q\) 个序列生成，当且仅当 \(a_i\) 在第 \(i\) 个序列中出现。你需要计数有多少个长度为 \(n\) 的，值域为 \([1,m]\) 的序列，使得任何一个可以被生成的序列都是这个序列的子序列。 \(n\leq 2\times 10^5,m\leq 10^8\)。

    我们考虑 \(q\) 个序列每个长度都为 \(1\) 的情况，那么就是说我们要计数以一个序列作为子序列的序列个数。

    设 \(f_i=(m-1)^{i-1}\) 表示花费 \(i\) 个位置匹配上一个元素的方案数，那么最终答案就是 \([n]f^q\)。

    如果某个序列长度为 \(2\)，假设它是 ab，那么一个序列合法，它不一定要匹配 ab，匹配 ba 也是可以的。所以，如果花费了 \(i\) 个位置匹配 a，那么前面 \(i-1\) 个位置不能是 a 也不能是 b，所以设 \(g_i=(m-2)^{i-1}\)，答案就是 \([n]f\times g\)。

    由此得到，设 \(f_1=f,f_2=g,f_3,f_4,f_5\) 以此类推，那么答案就相当于 \([n]\prod_{i=1}^5 f_i^{cnt_i}\)，其中 \(cnt_i\) 为长度 \(\geq i\) 的序列个数。

    直接做需要做 \(O(\log)\) 次 NTT，不太能接受。我们发现其实 \(f_i^{cnt_i}\) 可以组合意义求解：\(i\) 个数内选 \(cnt_i\) 个，最后一个必选，不选的有权值，总权值和。于是就只用计算 \(4\) 次 NTT，就做完了。

    code 
    //CF2169F
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const ll P = 998244353;
const ll G = 3;
int n;
vector<int> tr;

ll qp(ll x, ll y){
    ll ans = 1;
    while(y){
        if(y & 1){
            ans = ans * x % P;
        }
        x = x * x % P;
        y >>= 1;
    }
    return ans;
}
const ll iG = qp(G, P - 2);

int cc = 0;
inline void ntt(vector<ll> &f, bool op){
    ++ cc;
    for(int i = 0; i < n + n; ++ i){
        if(i < tr[i]){
            swap(f[i], f[tr[i]]);
        }
    }
    for(int p = 2; p <= n + n; p <<= 1){
        int len = p >> 1;
        ll tg = qp(op ? G : iG, (P-1) / p);
        for(int k = 0; k < n + n; k += p){
            ll buf = 1;
            for(int l = k; l < k + len; ++ l){
                ll tmp = buf * f[l+len] % P;
                f[l+len] = (f[l] - tmp);
                if(f[l+len] < 0) f[l+len] += P;
                f[l] = (f[l] + tmp);
                if(f[l] >= P) f[l] -= P;
                buf = buf * tg % P;
            }
        }
    }
}
vector<ll> operator * (vector<ll> p, vector<ll> q){
    for(int i = n; i < n + n; ++ i){
        p[i] = q[i] = 0;
    }
    ntt(p, 1);
    ntt(q, 1);
    vector<ll> res(n + n);
    for(int i = 0; i < n + n; ++ i){
        res[i] = p[i] * q[i] % P;
    }
    ntt(res, 0);
    ll val = qp(n + n, P - 2);
    for(int i = 0; i < n + n; ++ i){
        res[i] = res[i] * val % P;
    }
    return res;
}
vector<ll> qp(vector<ll> x, int y){
    vector<ll> ans(x.size());
    ans[0] = 1;
    while(y){
        if(y & 1){
            ans = ans * x;
        }
        x = x * x;
        y >>= 1;
    }
    return ans;
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    srand(0);
    int m, k;
    cin >> n >> m >> k;
    int act = n;
    n = 1;
    while(n < act + 1){
        n <<= 1;
    }
    vector<ll> fac(n), inv(n);
    fac[0] = 1;
    for(int i = 1; i < n; ++ i){
        fac[i] = fac[i-1] * i % P;
    }
    inv[n-1] = qp(fac[n-1], P-2);
    for(int i = n - 2; i >= 0; -- i){
        inv[i] = inv[i+1] * (i+1) % P;
    }
    auto C = [=](int x, int y){
        return (x == y && y == -1) ? 1 : (x < 0 || y < 0 || x < y) ? 0 : fac[x] * inv[y] % P * inv[x-y] % P;
    };
    tr.resize(n + n);
    for(int i = 0; i < n + n; ++ i){
        tr[i] = (tr[i>>1] >> 1) | ((i & 1) ? n + n >> 1 : 0);
    }
    vector<int> cnt(6);
    while(k--){
        int x;
        cin >> x;
        ++ cnt[x];
    }
    vector<vector<ll> > poly(6, vector<ll>(n + n));
    for(int i = cnt[1]+cnt[2]+cnt[3]+cnt[4]+cnt[5]; i < n; ++ i){
        poly[1][i] = qp(m-1, i-cnt[1]-cnt[2]-cnt[3]-cnt[4]-cnt[5]) * C(i-1, cnt[1]+cnt[2]+cnt[3]+cnt[4]+cnt[5]-1) % P;
    }
    for(int i = cnt[2]+cnt[3]+cnt[4]+cnt[5]; i < n; ++ i){
        poly[2][i] = qp(m-2, i-cnt[2]-cnt[3]-cnt[4]-cnt[5]) * C(i-1, cnt[2]+cnt[3]+cnt[4]+cnt[5]-1) % P;
    }
    for(int i = cnt[3]+cnt[4]+cnt[5]; i < n; ++ i){
        poly[3][i] = qp(m-3, i-cnt[3]-cnt[4]-cnt[5]) * C(i-1, cnt[3]+cnt[4]+cnt[5]-1) % P;
    }
    for(int i = cnt[4]+cnt[5]; i < n; ++ i){
        poly[4][i] = qp(m-4, i-cnt[4]-cnt[5]) * C(i-1, cnt[4]+cnt[5]-1) % P;
    }
    for(int i = cnt[5]; i < n; ++ i){
        poly[5][i] = qp(m-5, i-cnt[5]) * C(i-1, cnt[5]-1) % P;
    }
    vector<ll> ans(n + n);
    ans[0] = 1;
    ans = ans * poly[1] * poly[2] * poly[3] * poly[4] * poly[5];
    ll res = 0, now = 1;
    for(int i = act; i >= 0; -- i){
        res = (res + ans[i] * now) % P;
        now = now * m % P;
    }
    res = res * qp(2, cnt[2]) % P * qp(6, cnt[3]) % P * qp(24, cnt[4]) % P * qp(120, cnt[5]) % P;
    cout << res << '\n';
    return 0;
}
