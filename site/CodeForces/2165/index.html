    No.2165/2166 Div1 + Div2

    Contest Link: https://codeforces.com/contest/2165。

    2A. Same Difference¶

    计算有多少个字符和 \(s_n\) 不同。

    code 
    //CF2166A
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int T, n;
    cin >> T;
    while(T --){
        cin >> n;
        string s;
        cin >> s;
        int cnt = 0;
        for(int i = 0; i < n; ++ i){
            if(s[i] != s[n-1]){
                ++ cnt;
            }
        }
        cout << cnt << '\n';
    }
    return 0;
}

    2B. Tab Closing¶

    答案 \(\leq 2\)：先移动到最后一个位置，删到这个位置没有标签页后，移动到第一个标签页的位置。

    答案为 \(1\)，要么标签页太少导致刚开始每个标签页都是最大长度，要么标签页太宽，一个标签页都能顶满。

    code 
    //CF2166B
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int T, a, b, n;
    cin >> T;
    while(T --){
        cin >> a >> b >> n;
        if(a == b || 1ll * b * n <= a){
            cout << "1\n";
        } else {
            cout << "2\n";
        }
    }
    return 0;
}

    2C/1A. Cyclic Merging¶

    每次一定是找到最小的，合并到它两端更小的那一个上面去。链表维护。

    code 
    //CF2165A
#include <bits/stdc++.h>
using namespace std;

int main(){
    int T, n;
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin >> T;
    while(T--){
        cin >> n;
        vector<int> a(n), pre(n), nxt(n);
        vector<pair<int, int> > val(n);
        for(int i = 0; i < n; ++ i){
            cin >> val[i].first;
            a[i] = val[i].first;
            val[i].second = i;
            pre[i] = i-1;
            nxt[i] = i+1;
            if(i == 0){
                pre[i] = n-1;
            }
            if(i == n - 1){
                nxt[i] = 0;
            }
        }
        sort(val.begin(), val.end());
        long long ans = 0;
        for(int i = 0; i < n - 1; ++ i){
            int v = val[i].second;
            ans += min(a[pre[v]], a[nxt[v]]);
            pre[nxt[v]] = pre[v];
            nxt[pre[v]] = nxt[v];
        }
        cout << ans << '\n';
    }
    return 0;
}

    2D/1B. Marble Council¶

    考虑一个数字 \(i\) 出现了 \(cnt_i\) 次但可重集中没有 \(i\)，那么需要将这 \(cnt_i\) 全部隐藏在可重集中有的数的后面，就是说，设 \(S\) 为一个值的集合，那么合法当且仅当 \(\max_{i\notin S} cnt_i\leq \sum_{i\in S} cnt_i\)。方案数是 \(\prod_{i\in S} cnt_i\)。

    我们背包后取 \(f_{[mx,n]}\) 的和，其中 \(mx=\max_{i} cnt_i\) 即可。因为，\(f_i(i<mx)\) 不能使 \(cnt_i=mx\) 隐藏；\(f_i(i\geq mx)\) 无论 \(cnt_i=mx\) 选不选都是合法的。

    code 
    //CF2165B
#include <bits/stdc++.h>
using namespace std;
const long long P = 998244353;

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int T, n;
    cin >> T;
    while(T--){
        cin >> n;
        vector<int> a(n), cnt(n + 1);
        vector<long long> f(n+1);
        int mx = 0;
        for(int i = 0; i < n; ++ i){
            cin >> a[i];
            ++ cnt[a[i]];
            mx = max(mx, cnt[a[i]]);
        }
        long long ans = 0;
        f[0] = 1;
        for(int i = 1; i <= n; ++ i){
            for(int j = n; j >= cnt[i]; -- j){
                f[j] = (f[j] + f[j-cnt[i]] * cnt[i]) % P;
            }
        }
        for(int j = mx; j <= n; ++ j){
            ans = (ans + f[j]) % P;
        }
        cout << ans << '\n';
    }
    return 0;
}

    2E/1C. Binary Wine¶

    我们暴力地让最大的 \(a_i\) 去碰 \(c\) 的每一个位。具体地，维护一个大根堆，目前放了 \(a\) 所有数。设目前 \(c\) 的最高位为 \(k=2^a\)，弹出堆顶 \(x\)，那么如果 \(x \geq k\)，那么 \(c\) 的最高位显然可以达到，\(c\leftarrow c-k\) 后将 \(x-k\) 插入堆中；否则，我们需要让 \(x\) 够得着 \(k\)，答案加上 \(k-x\) 然后 \(c\leftarrow c-k\) 即可。

    最后，如果 \(c=0\)，那么直接输出答案；否则堆为空，答案需要再加上剩余的 \(c\)。

    多组询问的维护方式很简单，一个大根堆维护 \(a\)，新建另外一个大根堆维护每一轮新插入的 \(x-k\)，然后每次重置前一个大根堆，每次取堆顶就找两个堆更大的堆顶。

    code 
    //CF2165C
#include <bits/stdc++.h>
using namespace std;

int main(){
    int T;
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin >> T;
    while(T --){
        int n, q;
        cin >> n >> q;
        vector<int> a(n);
        priority_queue<int> qr;
        for(int i = 0; i < n; ++ i){
            cin >> a[i];
            qr.push(a[i]);
        }
        while(q --){
            vector<int> del;
            priority_queue<int> qq;
            int c;
            cin >> c;
            auto hb = [=](int c){
                return 1 << (31 ^ __builtin_clz(c));
            };
            int ans = 0;
            while(c && (qr.size() || qq.size())){
                int val;
                if(!qr.size() || (qq.size() && qq.top() > qr.top())){
                    val = qq.top();
                    qq.pop();
                } else {
                    val = qr.top();
                    qr.pop();
                    del.push_back(val);
                }
                if(val >= c){
                    c = 0;
                    break;
                } else if(hb(c) >= val){
                    ans += hb(c) - val;
                    c -= hb(c);
                } else {
                    val -= hb(c);
                    c -= hb(c);
                    qq.push(val);
                }
            }
            if(c){
                ans += c;
            }
            cout << ans << '\n';
            for(auto i : del){
                qr.push(i);
            }
        }
    }
    return 0;
}

    2F/1D. Path Split¶

    题意

    给定序列 \(a\)，将序列 \(a\) 拆分成若干个子序列，使得每个子序列内相邻两项差为 \(1\)，问最少能够拆分成多少个子序列。 \(n\leq 10^6\)。

    可以转化为最小链覆盖问题！两个点之间可以连边当且仅当差为 \(1\)。

    观察：每条能够连的边的贡献是等价的，所以经过调整可以证明，我们可以先考虑 \(1,2\) 之间的边，再考虑 \(2,3\) 之间的，再是 \(3,4\)，等等。

    直接这么做就是对的。一个更好写的做法：对于 \(i\in[1,2n]\)，从后往前找所有的 \(i\)，然后找它后继 \(i-1\)，如果存在且可以连边就连边；否则找它后继 \(i+1\)，如果存在且可以连边就连边。

    注释里有一些思考。

    code 
    //CF2165D
#include <bits/stdc++.h>
using namespace std;

int main(){
    int T;
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin >> T;
    while(T--){
        int n;
        cin >> n;
        vector<int> a(n), p(n);
        vector<set<int> > st(n*2);
        for(int i = 0; i < n; ++ i){
            cin >> a[i];
            -- a[i];
            p[i] = i;
            st[a[i]].insert(i);
        }
        auto cmp = [&](int x, int y) -> bool{
            return a[x] == a[y] ? x > y : a[x] < a[y];
        };
        int ans = n;
        sort(p.begin(), p.end(), cmp);
        for(int i = 0; i < n; ++ i){
            int pos = p[i];
            bool flg = 1;
            if(a[pos] != 0){
                auto pp = st[a[pos]-1].lower_bound(pos);
                if(pp != st[a[pos]-1].end()){
                    -- ans;
                    st[a[pos]-1].erase(pp);
                    flg = 0;
                }
            }
            if(flg && a[pos] != n + n - 1){
                auto pp = st[a[pos]+1].lower_bound(pos);
                if(pp != st[a[pos]+1].end()){
                    -- ans;
                    st[a[pos]+1].erase(pp);
                }
            }
        }
        cout << ans << '\n';
    }
    return 0;
}

/*
同一层，如果从左往右处理，那么每个点找前驱；右往左处理，每个点找后继。
反过来的话 hack:
1
6
2 3 2 4 2 1
正确的是3，错误的是4

不同层，如果从小到大处理，那么每一层值域相同的先连小边，再连大边

如果不是按出点分层，而是按边上小的那个点分层，可能不用考虑上述情况。
*/

    1E. Rainbow Branch¶

    题意

    给定一棵树，定义一个树的边 \(k\) 染色的权值为：所有简单路径中，颜色数最多的一条路径的颜色数。对于 \(k=1,2,...,n-1\)，求所有边 \(k\) 染色方案的最小权值。 \(3\leq n\leq 3\times 10^5\)。

    考虑对于每一个最小权值，求出最大的 \(k\)。

    观察 1：一定存在一个最优方案，使得每种颜色的边导出子图为恰好一个连通块。感性理解也是对的。

    那么，考虑建圆方树，那么它会有一个“中心”。

    如果权值是奇数，那么这个中心会是一个方点，这个方点到任何一个方点之间的方点数 \(\leq \dfrac{n+1}2\)；否则是个圆点，这个圆点到任何一个方点之间的方点数 \(\leq \dfrac n2\)。

    观察 2：为了让方点数量更多，那么每个方点都要尽可能地远离中心。

    说着抽象，形象一点描述，就是：

    • 对于奇数权值 \(2p+1\)，每次剥叶子，剥 \(p\) 层，剥掉的边两两染不同颜色，最后一整块一个颜色。
    • 对于偶数权值 \(2p+2\)，每次剥叶子，剥 \(p\) 层，剥掉的边两两染不同颜色，最后一整块找一个度数最大的点，这个点每个子树染一种颜色。

    维护是不难做到 \(O(n\log n)\) 的。

    code 
    #include <bits/stdc++.h>
using namespace std;

const int N = 3e5 + 10;
int T, n;

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin >> T;
    while(T--){
        cin >> n;
        vector<int> deg(n), out(n), degc(n + 1);
        for(int i = 1; i < n; ++ i){
            int u, v;
            cin >> u >> v;
            -- u;
            -- v;
            ++ deg[u];
            ++ deg[v];
            out[u] ^= v;
            out[v] ^= u;
        }
        queue<int> q;
        int mx = 0, la = 0, now = 0;
        for(int i = 0; i < n; ++ i){
            ++ degc[deg[i]];
            mx = max(mx, deg[i]);
            if(deg[i] == 1){
                q.push(i);
            }
        }
        for(int i = 1; i < n; ++ i){
            int gl = now;
            if(i & 1){
                gl += mx ? 1 : 0;
            } else {
                gl += mx;
            }
            // cout << i << ' ' << gl << ' ' << now << '\n';
            for(int j = la; j < gl; ++ j){
                cout << i << ' ';
            }
            la = gl;
            if(i % 2 == 0){
                int siz = q.size();
                for(int k = 0; k < siz; ++ k){
                    int x = q.front();
                    q.pop();
                    int y = out[x];
                    if(!deg[x]){
                        continue;
                    }
                    -- degc[1];
                    // cout << x << "->" << y << '\n';
                    ++ now;
                    out[y] ^= x;
                    -- degc[deg[y]];
                    -- deg[y];
                    ++ degc[deg[y]];
                    if(deg[y] == 1){
                        q.push(y);
                    }
                }
                // cout << degc[1] << '\n';
                while(mx >= 1 && !degc[mx]){
                    -- mx;
                }
            }
        }
        cout << '\n';
    }
    return 0;
}

    1F. Arctic Acquisition¶

    题意

    给你一个排列，问有多少个子区间满足：这个子区间内能够选出一个子序列，子序列内 \(5\) 个元素的大小排名是 21435。 \(n\leq 10^6\)。

    神秘 21435 题目。

    考虑对于每个 2，求出 21435 的 5 下标最小为多少。

    枚举 2，1 是可以贪心选的。

    枚举 3，考虑什么样的 435 最优，45 肯定是 3 两侧大于它的值，结论是 5 一定是 3 右侧第一个大于它的数，如果 3 到 5 之间还有一个 x，那么 4x5 和 43x 一定有一个存在且不劣；如果一个 43x 可以但是 435 不行（因为 4 的值大于 5 的值了），那么 45x 一定更优。

    于是 435 也只有 \(O(n)\) 组，找到这样的 21，435 组后二维数点即可。

    code 
    //CF2165F
#include <bits/stdc++.h>
using namespace std;

/*
拆分成 21 + 435
枚举 2，1 贪心找
然后支配点对？！枚举 3，考虑什么样的 435 最优。
可以发现如果 3,5 中间有一个 >3 的数 x，要么 43x，要么 4x5。
所以 5 是 3 后面第一个 > 它的数，然后去找 4，最后数点即可？！

代码里反一下 53412 好看一点
*/

const int N = 1e6 + 10;
int n, a[N], st[N], tp;
int mx[N], mn[N];

int t[N*4];
void add(int p, int l, int r, int x, int v){
    if(l == r){
        t[p] = max(t[p], v);
    } else {
        int mid = l + r >> 1;
        if(x <= mid){
            add(p<<1, l, mid, x, v);
        } else {
            add(p<<1|1, mid+1, r, x, v);
        }
        t[p] = max(t[p<<1], t[p<<1|1]);
    }
}
int ask(int p, int l, int r, int ql, int qr){
    if(qr < l || r < ql){
        return 0;
    } else if(ql <= l && r <= qr){
        return t[p];
    } else {
        int mid = l + r >> 1;
        return max(ask(p<<1, l, mid, ql, qr), ask(p<<1|1, mid+1, r, ql, qr));
    }
}

int bit[N];
void add(int x, int v){
    while(x <= n){
        bit[x] = min(bit[x], v);
        x += x & -x;
    }
}
int ask(int x){
    int mn = n + 1;
    while(x){
        mn = min(mn, bit[x]);
        x -= x & -x;
    }
    return mn;
}

int main(){
    int T;
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    cin >> T;
    while(T --){
        cin >> n;
        for(int i = 1; i <= n; ++ i){
            cin >> a[i];
            bit[i] = n + 1;
            t[i] = t[i+n] = t[i+n+n] = t[i+n+n+n] = 0;
        }
        reverse(a + 1, a + n + 1);
        tp = 0;
        for(int i = 1; i <= n; ++ i){
            while(tp && a[st[tp]] < a[i]){
                -- tp;
            }
            mx[i] = st[tp];
            st[++tp] = i;
        }
        tp = 0;
        for(int i = 1; i <= n; ++ i){
            while(tp && a[st[tp]] > a[i]){
                -- tp;
            }
            mn[i] = st[tp];
            st[++tp] = i;
        }
        vector<vector<int> > hav(n+1), qry(n+1);
        for(int i = n; i >= 1; -- i){
            if(mn[i]){
                hav[mn[i]].push_back(i);
            }
        }
        for(int i = n; i >= 1; -- i){
            if(mx[i]){
                //找 i 后面下标最小的 < mx[i] 的值
                int R = ask(1, 1, n, a[i], a[mx[i]]);
                if(R){
                    qry[n-R+1].push_back(i);
                }
            }
            add(1, 1, n, a[i], n - i + 1);
        }
        vector<int> ans(n + 1);
        for(int i = n; i >= 1; -- i){
            for(auto j : qry[i]){
                int R = ask(a[j]), L = mx[j];
                if(L != 0 && R != n + 1){
                    ans[R] = max(ans[R], L);
                }
            }
            for(auto j : hav[i]){
                add(a[j], j);
            }
        }
        long long rs = 0;
        int le = 0;
        for(int i = 1; i <= n; ++ i){
            le = max(le, ans[i]);
            rs += le;
        }
        cout << rs << '\n';
    }
    return 0;
}
