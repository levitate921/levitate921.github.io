    A. Souvlaki VS. Kalamaki¶

    排好序后，检查偶数轮交换的两个位置是否均相同即可。

    code 
    //CF2163A
#include <bits/stdc++.h>
using namespace std;

const int N = 110;
int T, n, a[N];

int main(){
    scanf("%d", &T);
    while(T--){
        scanf("%d", &n);
        for(int i = 1; i <= n; ++ i){
            scanf("%d", &a[i]);
        }
        sort(a + 1, a + n + 1);
        bool flg = 1;
        for(int i = 2; i + 1 <= n; i += 2){
            if(a[i] != a[i+1]){
                flg = 0;
            }
        }
        puts(flg ? "YES" : "NO");
    }
    return 0;
}

    B. Siga ta Kymata¶

    很难的 B。尝试 dp，会发现每次操作区间不一定不交。

    正解是，如下五次操作：\((1,pos_1),(pos_1,n),(1,pos_n),(pos_n,n),(pos_1,pos_n)\)，如果 \(1\) 的位置在 \(n\) 的位置后，那么最后一个换成 \((pos_n,pos_1)\)。

    如果操作完合法就行，否则就不行。

    code 
    //CF2163B
#include <bits/stdc++.h>
using namespace std;

const int N = 2e5 + 10;
int T, n, a[N];
int p[N];
char s[N];

void calc(int l, int r){
    for(int i = l + 1; i < r; ++ i){
        if(min(a[l], a[r]) < a[i] && a[i] < max(a[l], a[r])){
            p[i] = 1;
        }
    }
}

int main(){
    scanf("%d", &T);
    while(T--){
        scanf("%d", &n);
        int mn, mx;
        for(int i = 1; i <= n; ++ i){
            scanf("%d", &a[i]);
            if(a[i] == 1){
                mn = i;
            } else if(a[i] == n){
                mx = i;
            }
            p[i] = 0;
        }
        scanf("%s", s + 1);
        calc(1, mn);
        calc(mn, n);
        calc(1, mx);
        calc(mx, n);
        calc(min(mn, mx), max(mn, mx));
        bool flg = 1;
        for(int i = 1; i <= n; ++ i){
            if(s[i] == '1' && !p[i]){
                flg = 0;
                break;
            }
        }
        if(flg){
            puts("5");
            printf("%d %d\n", 1, mn);
            printf("%d %d\n", mn, n);
            printf("%d %d\n", 1, mx);
            printf("%d %d\n", mx, n);
            printf("%d %d\n", min(mn, mx), max(mn, mx));
        } else {
            puts("-1");
        }
    }
    return 0;
}

    C. Monopati¶

    我们其实可以预处理所有路径的 \(\min,\max\) 值。。。然后扫描线扫一下就行，比 B 简单。

    code 
    //CF2163C
#include <bits/stdc++.h>
using namespace std;

const int N = 4e5 + 10;
int T, n, a[2][N];
int pmx[N], pmn[N], smx[N], smn[N];
int tt[N];

int main(){
    scanf("%d", &T);
    while(T --){
        scanf("%d", &n);
        for(int i = 1; i <= n; ++ i){
            tt[i] = tt[i+n] = 0;
            scanf("%d", &a[0][i]);
            if(i == 1){
                pmx[i] = pmn[i] = a[0][i];
            } else {
                pmx[i] = max(pmx[i-1], a[0][i]);
                pmn[i] = min(pmn[i-1], a[0][i]);
            }
        }
        for(int i = 1; i <= n; ++ i){
            scanf("%d", &a[1][i]);
        }
        for(int i = n; i >= 1; -- i){
            if(i == n){
                smx[i] = smn[i] = a[1][i];
            } else {
                smx[i] = max(smx[i+1], a[1][i]);
                smn[i] = min(smn[i+1], a[1][i]);
            }
        }
        for(int i = 1; i <= n; ++ i){
            int L = min(pmn[i], smn[i]);
            int R = max(pmx[i], smx[i]);
            tt[R] = max(tt[R], L);
            // printf("%d %d\n", L, R);
        }
        long long cnt = 0;
        int le = 0;
        for(int i = 1; i <= n + n; ++ i){
            le = max(le, tt[i]);
            cnt += le;
        }
        printf("%lld\n", cnt);
    }
    return 0;
}

    D1/D2. Diadrash¶

    题意

    交互题，给你 \(q\) 若干个区间，交互库有一个你不知道的排列 \(p\)，每次任意指定询问区间 \([l,r]\)，返回区间的 \(\operatorname{mex}\)，最后要找到这 \(q\) 个区间内最大 \(\operatorname{mex}\) 的值。 简单版：\(300\) 次询问，困难版：\(30\) 次询问。\(4\leq n\leq 10^4\)。

    观察 1：如果一个区间完全被另一个区间包含，前者的 \(\operatorname{mex}\) 严格小于后者，是没用的。

    观察 2：\(\operatorname{mex}_{i\in[l,r]}\{a_i\}=\min(\operatorname{mex}_{i\in[1,r]}\{a_i\},\operatorname{mex}_{i\in[l,n]}\{a_i\})\)。证明不难。

    由观察 2，设区间 \([l,r]\) 的 \(\operatorname{mex}\) 为 \(k\)，如果我们查到了 \([1,r],[l,n]\) 的 \(\operatorname{mex}\)，且前者的小于后者的。那么对于区间 \([l',r'],l'<l,r'<r\)，\([l',r']\) 的答案是不可能更优的，因为序列中的元素 \(k\) 一定在 \((r,n]\) 之中，且这个区间与 \([l',r']\) 无交。

    同理，如果 \([1,r],[l,n]\) 的 \(\operatorname{mex}\) 均为 \(k\)，那么这个区间 \([l,r]\) 直接就是最优解了。

    所以我们可以二分区间：将区间按照左端点排序，由观察 1 删掉若干无用区间，然后剩下区间两端点都是排好序的，在这个区间序列上进行二分，二分收缩方向由 \([1,r],[l,n]\) 的大小关系决定。总询问次数 \(2\log n\)。

    code 
    //CF2163D2
#include <bits/stdc++.h>
using namespace std;

const int N = 3e5 + 10;
int T, n, q;
pair<int, int> a[N];

bool cmp(pair<int, int> x, pair<int, int> y){
    if(x.first != y.first){
        return x.first < y.first;
    }
    return x.second > y.second;
}
int ask(int x, int y){
    printf("? %d %d\n", x, y);
    fflush(stdout);
    scanf("%d", &x);
    return x;
}

int main(){
    scanf("%d", &T);
    while(T--){
        scanf("%d%d", &n, &q);
        for(int i = 1; i <= q; ++ i){
            scanf("%d%d", &a[i].first, &a[i].second);
        }
        sort(a + 1, a + q + 1, cmp);
        int L = 0;
        int now = 0;
        for(int i = 1; i <= q; ++ i){
            // printf("- %d %d\n", a[i].first, a[i].second, now);
            if(a[i].second > now){
                now = a[i].second;
                a[++L] = a[i];
            }
        }
        int l = 1, r = L, ans = 0;
        while(l <= r){
            int mid = l + r >> 1;
            int le = ask(a[mid].first, n);
            int ri = ask(1, a[mid].second);
            ans = max(ans, min(le, ri));
            if(le < ri){
                r = mid - 1;
            } else {
                l = mid + 1;
            }
        }
        printf("! %d\n", ans);
        fflush(stdout);
    }
    return 0;
}

    E. Plegma¶

    题意

    通信题，对于 Alice，给一个 \(n*n\) 的 01 矩阵，Alice 要返回两个 \([1,n]\) 之内的数 \(i,j\)；对于 Bob，它接收到 01 矩阵的第 \(i\) 行，第 \(j\) 列的具体值，但是不知道 \(i,j\) 的值，Bob 需要返回这个 01 矩阵中所有 1 是否构成一个四连通块。

    通信传递的信息量其实是远大于答案的信息量的，所以我们考虑，能否通过某一位传递信息？

    定义 Bob 收到的两个字符串为行串和列串。

    如果约定 Bob 收到的列串第一个字符为 \(1\)，那么就是连通，否则不连通，这样可以吗？

    发现，会出现这种情况：第一行全是 \(1\)，且不连通，这样的话，Alice 无法传递一个第一个字符不为 \(1\) 的列串。

    那么此时会出现一个更加显著的特征：Alice 可以传递一个全为 \(1\) 的行串！同理，第一行全为 \(0\) 也可以传递全为 \(0\) 的行串。

    最后的问题是，如何区分第一行全为 \(0\) 但连通/不连通？

    我们只需要换一个方向，就是用 \(1\) 表示不连通，\(0\) 表示连通即可。

    具体地：

    第一行全1： 连通：全1行 + 任意列 不连通：任意非全1行 + 任意列

    第一行全0： 连通：任意非全0行 + 任意列 不连通：全0行 + 任意列

    第一行有0有1： 连通：第一行 + 第一位0的列 不连通：第一行 + 第一位1的列

    code 
    //CF2163E
#include <bits/stdc++.h>
using namespace std;

const int N = 1010;

int T, n, C;
char s[N][N];
char op[10];

int main(){
    scanf("%s", op);
    scanf("%d", &T);
    while(T --){
        if(op[0] == 'f'){
            scanf("%d%d", &n, &C);
            for(int i = 1; i <= n; ++ i){
                scanf("%s", s[i] + 1);
            }
            int c0 = 0, c1 = 0;
            for(int j = 1; j <= n; ++ j){
                if(s[1][j] == '0'){
                    c0 = j;
                } else {
                    c1 = j;
                }
            }
            if(c0 == 0){
                if(C == 1){
                    printf("%d %d\n", 1, 1);
                } else {
                    int hang = 0;
                    for(int i = 2; i <= n; ++ i){
                        bool flg = 0;
                        for(int j = 1; j <= n; ++ j){
                            if(s[i][j] == '0'){
                                flg = 1;
                                break;
                            }
                        }
                        if(flg){
                            hang = i;
                            break;
                        }
                    }
                    printf("%d %d\n", hang, 1);
                }
            } else if(c1 == 0){
                if(C == 0){
                    printf("%d %d\n", 1, 1);
                } else {
                    int hang = 0;
                    for(int i = 2; i <= n; ++ i){
                        bool flg = 0;
                        for(int j = 1; j <= n; ++ j){
                            if(s[i][j] == '1'){
                                flg = 1;
                                break;
                            }
                        }
                        if(flg){
                            hang = i;
                            break;
                        }
                    }
                    printf("%d %d\n", hang, 1);
                }
            } else {
                for(int j = 1; j <= n; ++ j){
                    if(s[1][j] - '0' != C){
                        printf("%d %d\n", 1, j);
                        break;
                    }
                }
            }
        } else {
            scanf("%d", &n);
            scanf("%s%s", s[1] + 1, s[2] + 1);
            int c0 = 0, c1 = 0;
            for(int i = 1; i <= n; ++ i){
                if(s[1][i] == '0'){
                    c0 = i;
                } else if(s[1][i] == '1'){
                    c1 = i;
                }
            }
            if(c0 == 0){
                puts("1");
            } else if(c1 == 0){
                puts("0");
            } else {
                if(s[2][1] == '0'){
                    puts("1");
                } else {
                    puts("0");
                }
            }
        }
    }
    return 0;
}
