    A. Sequence Game¶

    检验是否有 \(\min a\leq x \leq \max a\) 即可。

    code 
    //CF2164A
#include <bits/stdc++.h>
using namespace std;

const int N = 110;
int n, a[N], x;

int main(){
    int T;
    scanf("%d", &T);
    while(T--){
        scanf("%d", &n);
        for(int i = 1; i <= n ;++ i){
            scanf("%d", &a[i]);
        }
        scanf("%d", &x);
        sort(a + 1, a + n + 1);
        if(x < a[1] || x > a[n]){
            puts("No");
        } else {
            puts("Yes");
        }
    }
    return 0;
}

    B. Even Modulo Pair¶

    暴力复杂度是正确的！考虑 -1 的条件：有两个偶数不行，有两个奇数，大的小于小的两倍不行，所以，-1 的序列长度为 \(O(\log V)\) 的。

    然后你直接暴力，或者多随几组就能过。

    code 
    //CF2164B
#include <bits/stdc++.h>
using namespace std;

const int N = 1e5 + 10;
int T, n, a[N];

int main(){
    scanf("%d", &T);
    while(T--){
        scanf("%d", &n);
        for(int i = 1; i <= n; ++ i){
            scanf("%d", &a[i]);
        }
        sort(a + 1, a + n + 1);
        bool flg = 1;
        for(int i = 1; i <= 500; ++ i){
            int x = rand() % n + 1;
            int y = rand() % n + 1;
            if(x > y) swap(x, y);
            if(x < y && (a[y] % a[x]) % 2 == 0){
                flg = 0;
                printf("%d %d\n", a[x], a[y]);
                break;
            }
        }
        if(flg){
            puts("-1");
        }
    }
    return 0;
}

    C. Dungeon¶

    有贪心的打怪策略：先打 \(b\leq c\) 的，这部分按照 \(b\) 从小往大打，每次选能打赢的攻击力最低的武器；再打 \(b\geq c\) 的，怎么打都行，反正没损耗；最后打 \(c=0\) 的，每次选能打赢的攻击力最低的武器，因为要消耗掉武器。

    code

    ```cpp //CF2164C

    include ¶

    using namespace std;

    const int N = 2e5 + 10; int T, n, m, b[N], c[N]; vector > up; vector dn, zr;

    int main(){ scanf("%d", &T); while(T--){ scanf("%d%d", &n, &m); multiset st; vector > ().swap(up); vector ().swap(dn); vector ().swap(zr); for(int i = 1; i <= n; ++ i){ int x; scanf("%d", &x); st.insert(x); } for(int i = 1; i <= m; ++ i){ scanf("%d", &b[i]); } for(int i = 1; i <= m; ++ i){ scanf("%d", &c[i]); if(c[i] == 0){ zr.push_back(b[i]); } else if(c[i] <= b[i]){ dn.push_back(b[i]); } else { up.push_back(make_pair(b[i], c[i])); } } sort(up.begin(), up.end()); sort(zr.begin(), zr.end()); int cnt = 0; for(auto i : up){ int x = i.first; if(st.lower_bound(x) != st.end()){ auto it = st.lower_bound(x); int val = max(*it, i.second); st.erase(it); st.insert(val); ++ cnt; } } for(auto i : dn){ if(st.lower_bound(i) != st.end()){ ++ cnt; } } for(auto i : zr){ if(st.lower_bound(i) != st.end()){ auto it = st.lower_bound(i); st.erase(it); ++ cnt; } } printf("%d\n", cnt); } return 0; } ````

    D. Copy String¶

    二分，然后双指针，第一个串中位置 \(i\) 可以匹配第二个串中 \([i,i+r]\) 中的某一段，从左往右贪心匹配就行。

    code 
    //CF2164D
#include <bits/stdc++.h>
using namespace std;

const int N = 1e6 + 10;
int T, n, K;
char s[N], t[N];
int rr[N];

bool chk(int len){
    int j = 0;
    for(int i = 1; i <= n; ++ i){
        if(i - 1 > j){
            return 0;
        }
        while(j < n && t[j+1] == s[i] && j - i < len){
            ++ j;
        }
        rr[i] = j;
    }
    return j == n;
}

int main(){
    scanf("%d", &T);
    while(T--){
        scanf("%d%d", &n, &K);
        scanf("%s%s", s + 1, t + 1);
        if(!chk(K)){
            puts("-1");
        } else {
            int l = 0, r = K;
            while(l < r){
                int mid = l + r >> 1;
                if(chk(mid)){
                    r = mid;
                } else {
                    l = mid + 1;
                }
            }
            chk(l);
            printf("%d\n", l);
            // for(int i = 1; i <= n; ++ i){
            //     printf("%d ", rr[i]);
            // }puts("");
            for(int i = 1; i <= l; ++ i){
                // int cc = 0;
                for(int j = 1; j <= n; ++ j){
                    for(int k = min(rr[j-1], j + i - 1) + 1; k <= min(rr[j], j + i); ++ k){
                        putchar(s[j]);
                        // ++ cc;
                    }
                }
                // for(int j = cc + 1; j <= n; ++ j){
                //     putchar(s[j]);
                // }
                putchar('\n');
            }
        }
    }
    return 0;
}

    E. Journey¶

    问题转化为，对于所有奇数度数点之间跑一组匹配，边权为按照题意这两点之间编号最大的边边权最小能是多少，然后求最小匹配大小。

    考虑按照边的编号从小到大建 kruskal 重构树，那么两个点连边的最小边权就相当于它们的 lca 的根链上的最小边权。然后我们建出树后 dfs 一遍将父边的边权下传下去，这样的话，每两个奇数点一定在 lca 处匹配掉，所以一条边有贡献，当且仅当它子树内奇数点个数为奇数。精细实现一下可以写的很短。

    code 
    //CF2164E
#include <bits/stdc++.h>
using namespace std;

const int N = 2e6 + 10;
int T, n, m;
int eg[N][3];
int deg[N], fa[N], val[N], nd[N];
int fae[N];

int gf(int x){
    return x == fa[x] ? x : fa[x] = gf(fa[x]);
}

int main(){
    scanf("%d", &T);
    while(T --){
        scanf("%d%d", &n, &m);
        for(int i = 1; i <= n + n; ++ i){
            deg[i] = 0;
            fa[i] = i;
            nd[i] = 0;
            val[i] = 1e9;
            fae[i] = 0;
        }
        long long ans = 0;
        for(int i = 1; i <= m; ++ i){
            int u, v, w;
            scanf("%d%d%d", &u, &v, &w);
            eg[i][0] = u;
            eg[i][1] = v;
            eg[i][2] = w;
            deg[u] ^= 1;
            deg[v] ^= 1;
            ans += w;
        }
        int cnt = n;
        for(int i = 1; i <= m; ++ i){
            int u = gf(eg[i][0]);
            int v = gf(eg[i][1]);
            int w = eg[i][2];
            if(u == v){
                val[u] = min(val[u], w);
            } else {
                ++ cnt;
                if(deg[u] && deg[v]){
                    nd[cnt] = 1;
                }
                deg[cnt] = deg[u] ^ deg[v];
                fa[u] = fa[v] = cnt;
                fae[u] = fae[v] = cnt;
                val[cnt] = w;
            }
        }
        for(int i = n + n - 1; i >= 1; -- i){
            if(fae[i]){
                val[i] = min(val[i], val[fae[i]]);
            }
            if(nd[i]){
                ans += val[i];
            }
        }
        printf("%lld\n", ans);
    }
    return 0;
}

    F1/F2. Chain Prefix Rank¶

    题意

    给你一棵树，每个点有点权 \(a\)，计数有多少个排列 \(p\)，使得对于所有 \(i\)，有 \(a_i\) 个 \(i\) 的祖先的 \(p\) 值小于 \(p_i\)。 简单版：\(n\leq 5000\)；困难版：\(n\leq 5\times 10^5\)。

    我们考虑计算一个叶子 \(i\)，考虑它的一条根链，显然这条链上每个点的两两大小关系是固定的（考虑从上到下加入每个点）。然后，考虑这个叶子节点的父亲 \(j\)，那么 \(p_i\) 的具体大小就没有用了！我们只关心它放在 \(j\) 的根链上这些点中哪两个点之间就行，那么对于 \(j\) 的不同子树，这个信息是可以合并的，然后合并过程中，两棵子树的部分合并起来会乘以一个组合数，然后把这些组合数全部乘起来就得到了答案。

    具体的，设 \(f_{i,j}\) 表示 \(i\) 子树内，有多少个点还需要在 \(i\) 的祖先中放 \(j\) 个小于它的点才能符合题意。合并的时候就是 \(f_{x,j}+f_{y,j}\to f_{i,j}\)，全局的答案乘上组合数 \(\dbinom {f_{x,j}+f_{y,j}}{f_{y,j}}\)。最后将 \(i\) 插入到状态中，就是合并 \(f_{i,a_i}\) 和 \(f_{i,a_{i+1}}\)，注意此时 \(i\) 一定是位于这两个中间，所以只有一种方案，最后是 \(a_{i+1}\) 后面的下标全体左移一位。

    直接计算复杂度 \(O(n^2)\)，可以通过 F1。优化是考虑启发式合并，对于每个点维护在第二维下标上的平衡树，然后维护 tag 表示下标的偏移量即可。复杂度 \(O(n\log^2 n)\)，可以通过 F2。

    //CF2164F1
#include <bits/stdc++.h>
using namespace std;

const int N = 5010;
int T, n, fa[N], a[N];
typedef long long ll;
const ll P = 998244353;
ll fac[N], inv[N];

ll qp(ll x, ll y){
    ll ans = 1;
    while(y){
        if(y & 1){
            ans = ans * x % P;
        }
        x = x * x % P;
        y >>= 1;
    }
    return ans;
}
ll C(int x, int y){
    if(x < 0 || y < 0 || x < y){
        return 0;
    }
    return fac[x] * inv[y] % P * inv[x-y] % P;
}

int f[N][N];
int dep[N];
vector<int> g[N];
ll ans;

void dfs(int x){
    if(!g[x].size()){
        f[x][a[x]] = 1;
        return;
    }
    for(int i : g[x]){
        dfs(i);
        for(int j = 0; j <= n; ++ j){
            ans = ans * C(f[x][j] + f[i][j], f[i][j]) % P;
            f[x][j] += f[i][j];
        }
    }
    for(int j = a[x]; j < n; ++ j){
        f[x][j] += f[x][j+1];
        f[x][j+1] = 0;
    }
    ++ f[x][a[x]];
}

int main(){
    n = 5000;
    fac[0] = 1;
    for(int i = 1; i <= n; ++ i){
        fac[i] = fac[i-1] * i % P;
    }
    inv[n] = qp(fac[n], P - 2);
    for(int i = n - 1; i >= 0; -- i){
        inv[i] = inv[i+1] * (i + 1) % P;
    }
    scanf("%d", &T);
    while(T--){
        scanf("%d", &n);
        ans = 1;
        for(int i = 1; i <= n; ++ i){
            vector<int> ().swap(g[i]);
            memset(f[i], 0, sizeof(f[i]));
        }
        for(int i = 2; i <= n; ++ i){
            scanf("%d", &fa[i]);
            dep[i] = dep[fa[i]] + 1;
            g[fa[i]].push_back(i);
        }
        for(int i = 1; i <= n; ++ i){
            scanf("%d", &a[i]);
        }
        dfs(1);
        printf("%lld\n", ans);
    }
    return 0;
}

    //CF2164F2
#include <bits/stdc++.h>
using namespace std;

const int N = 500010;
int T, n, fa[N], a[N];
typedef long long ll;
const ll P = 998244353;
ll fac[N], inv[N];

ll qp(ll x, ll y){
    ll ans = 1;
    while(y){
        if(y & 1){
            ans = ans * x % P;
        }
        x = x * x % P;
        y >>= 1;
    }
    return ans;
}
ll C(int x, int y){
    if(x < 0 || y < 0 || x < y){
        return 0;
    }
    return fac[x] * inv[y] % P * inv[x-y] % P;
}

vector<int> g[N];
ll ans;

struct node{
    int ls, rs, pos, tag, val, siz, pri;
} t[N*20];
int rt[N], tot;
int nnd(int x, int vl){
    ++ tot;
    t[tot].pos = x;
    t[tot].ls = t[tot].rs = t[tot].tag = 0;
    t[tot].val = vl;
    t[tot].siz = 1;
    t[tot].pri = rand();
    return tot;
}
void upd(int x){
    t[x].siz = t[t[x].ls].siz + 1 + t[t[x].rs].siz;
}
void psd(int x){
    if(!t[x].tag){
        return;
    }
    if(t[x].ls){
        t[t[x].ls].pos -= t[x].tag;
        t[t[x].ls].tag += t[x].tag;
    }
    if(t[x].rs){
        t[t[x].rs].pos -= t[x].tag;
        t[t[x].rs].tag += t[x].tag;
    }
    t[x].tag = 0;
}
int mg(int p, int q){
    psd(p);
    psd(q);
    if(!p || !q){
        return p + q;
    } else if(t[p].pri < t[q].pri){
        t[p].rs = mg(t[p].rs, q);
        upd(p);
        return p;
    } else {
        t[q].ls = mg(p, t[q].ls);
        upd(q);
        return q;
    }
}
void sp(int p, int k, int &x, int &y){
    if(!p){
        x = y = 0;
        return;
    }
    psd(p);
    if(t[p].pos <= k){
        x = p;
        sp(t[p].rs, k, t[p].rs, y);
    } else {
        y = p;
        sp(t[p].ls, k, x, t[p].ls);
    }
    upd(p);
}
void add(int &p, int pos, int val, int op){
    int x = 0, y = 0, z = 0;
    sp(p, pos, x, z);
    sp(x, pos-1, x, y);
    if(!y){
        y = nnd(pos, val);
    } else {
        if(op){
            ans = ans * C(val + t[y].val, val) % P;
        }
        t[y].val += val;
    }
    x = mg(x, y);
    p = mg(x, z);
}
void comb(int p, int q){
    if(!q){
        return;
    }
    psd(q);
    add(rt[p], t[q].pos, t[q].val, 1);
    comb(p, t[q].ls);
    comb(p, t[q].rs);
}
void weiy(int p, int pos){
    int x = 0, y = 0, z = 0;
    sp(rt[p], pos, x, z);
    sp(z, pos+1, y, z);
    if(z){
        t[z].tag += 1;
        t[z].pos -= 1;
    }
    if(y){
        add(x, t[y].pos - 1, t[y].val, 0);
    }
    rt[p] = mg(x, z);
}
void dfs(int x){
    if(!g[x].size()){
        rt[x] = nnd(a[x], 1);
        return;
    }
    for(int i : g[x]){
        dfs(i);
        if(t[rt[x]].siz < t[rt[i]].siz){
            comb(i, rt[x]);
            swap(rt[x], rt[i]);
        } else {
            comb(x, rt[i]);
        }
    }
    add(rt[x], a[x], 1, 0);
    weiy(x, a[x]);
}

int main(){
    srand(5);
    n = 500000;
    fac[0] = 1;
    for(int i = 1; i <= n; ++ i){
        fac[i] = fac[i-1] * i % P;
    }
    inv[n] = qp(fac[n], P - 2);
    for(int i = n - 1; i >= 0; -- i){
        inv[i] = inv[i+1] * (i + 1) % P;
    }
    scanf("%d", &T);
    while(T--){
        tot = 0;
        scanf("%d", &n);
        ans = 1;
        for(int i = 1; i <= n; ++ i){
            rt[i] = 0;
            vector<int> ().swap(g[i]);
        }
        for(int i = 2; i <= n; ++ i){
            scanf("%d", &fa[i]);
            g[fa[i]].push_back(i);
        }
        for(int i = 1; i <= n; ++ i){
            scanf("%d", &a[i]);
        }
        dfs(1);
        printf("%lld\n", ans);
    }
    return 0;
}
