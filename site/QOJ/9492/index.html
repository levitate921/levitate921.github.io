    2024 2B 树上简单求和

    Problem Link: https://qoj.ac/problem/9492。

    题意

    有 \(n\) 个节点，第 \(x\) 个节点有点权 \(a_x\)，有两棵树分别独立地连接了这 \(n\) 个点，两棵树点权是共用的。你需要进行 \(m\) 次操作，每次操作给定 \(x,y,k\)，进行以下两步： 1. 将第一棵树上 \(x,y\) 之间最短路径上所有节点的点权增加 \(k\)； 2. 求第二棵树上 \(x,y\) 之间最短路径上的点权和对 \(2^{64}\) 取模的结果。 \(n,m\leq 2\times 10^5\)。

    三种做法。

    Part 1. 树分块¶

    首先一条路径拆成四条根链。

    对两棵树都随机撒点分块，使得每个点向上走 \(O(B)\) 步都能走到一个关键点。

    1. T1 关键点根链 -> T2 关键点根链。预处理出贡献系数即可。
    2. T1 关键点根链 -> T2 单点。对 T1 进行 dfn 序分块，变为 dfn 序上的 \(O(B)\) 单点加 \(O(1)\) 区间查问题。
    3. T1 单点 -> T2 关键点根链。对 T2 进行 dfn 序分块，变为 dfn 序上的 \(O(B)\) 区间加 \(O(1)\) 单点查问题。
    4. T1 单点 -> T2 单点。直接映射就行。

    复杂度 \(O(n\sqrt n)\)。

    一个有一点变化的做法：T1 不分块，维护 T1 每个根链到 T2 关键点根链的贡献系数，T1 根链-> T2 单点一样是 dfn 序分块。需要逐块处理才能做到线性空间复杂度，而且常数感觉很大。

    Part 2. 树剖+分块¶

    链通过树剖拆成 \(O(\log n)\) 个区间，然后就变成了一个经典问题：\(a\) 区间加，查 \(a_{p_{[l,r]}}\) 的和。

    设左侧为 \([1,n]\)，右侧为 \(p_{[1,n]}\)。 对两侧都分块，维护两个大小为 \(n/B\) 的 \(left,right\) 数组，求出 \(cnt_{i,j}\) 表示左侧前 \(i\) 个块对右侧第 \(j\) 个块的贡献系数。

    1. 整块 -> 整块：通过 \(cnt\) 的值，修改 \(right\)。
    2. 整块 -> 单点：先不贡献到右侧，使用 \(left\) 记录即可。
    3. 单点 -> 整块：修改 \(right\)。
    4. 单点 -> 单点：直接记录即可。

    复杂度 \(O(n\log n\sqrt n)\)，非常快！

    Part 3. 括号序/欧拉序分块¶

    对两棵树求欧拉序，然后第一次出现贡献为 \(+1\)，第二次出现为 \(-1\)，那么，一条根链就相当于括号序中的一段前缀贡献的和。使用 Part 2 中的序列分块手法维护即可。复杂度 \(O(n\sqrt n)\)，不知道快不快。

    第二种做法的实现：

    //qoj9492
#include <bits/stdc++.h>
using namespace std;
typedef unsigned long long ll;

struct TREE{
    int n, dfn_cnt;
    vector<vector<int> > graph;
    vector<int> size, depth, son, dfn, top, father;
    void dfs(int x, int fa){
        size[x] = 1;
        father[x] = fa;
        for(auto i : graph[x]){
            if(i == fa){
                continue;
            }
            depth[i] = depth[x] + 1;
            dfs(i, x);
            size[x] += size[i];
            if(son[x] == -1 || size[i] > size[son[x]]){
                son[x] = i;
            }
        }
    }
    void dfs_again(int x, int topnode){
        dfn[x] = dfn_cnt ++;
        top[x] = topnode;
        if(son[x] != -1){
            dfs_again(son[x], topnode);
        }
        for(auto i : graph[x]){
            if(i == father[x] || i == son[x]){
                continue;
            }
            dfs_again(i, i);
        }
    }
    void read_init(int node_cnt){
        dfn_cnt = 0;
        n = node_cnt;
        graph.resize(n);
        size.resize(n);
        depth.resize(n);
        son.resize(n, -1);
        dfn.resize(n);
        top.resize(n);
        father.resize(n);
        for(int i = 1; i < n; ++ i){
            int u, v;
            cin >> u >> v;
            -- u;
            -- v;
            graph[u].push_back(v);
            graph[v].push_back(u);
        }
        dfs(0, 0);
        dfs_again(0, 0);
    }
    void decomp(int u, int v, vector<pair<int, int> > &result){
        result.clear();
        while(top[u] != top[v]){
            if(depth[top[u]] < depth[top[v]]){
                swap(u, v);
            }
            result.emplace_back(dfn[top[u]], dfn[u]);
            u = father[top[u]];
        }
        if(depth[u] > depth[v]){
            swap(u, v);
        }
        result.emplace_back(dfn[u], dfn[v]);
    }
};

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int n, q;
    cin >> n >> q;
    vector<ll> value(n);
    for(int i = 0; i < n; ++ i){
        cin >> value[i];
    }
    TREE tr1, tr2;
    tr1.read_init(n);
    tr2.read_init(n);

    const int B = 450;
    const int block_cnt = (n + B - 1) / B;
    vector<int> fr(n), to(n);
    vector<int> L(block_cnt), R(block_cnt);
    for(int i = 0; i < block_cnt; ++ i){
        L[i] = i * B;
        R[i] = min(L[i] + B, n) - 1;
    }
    for(int i = 0; i < n; ++ i){
        to[tr1.dfn[i]] = tr2.dfn[i];
        fr[tr2.dfn[i]] = tr1.dfn[i];
    }
    vector<vector<ll> > cnt(block_cnt, vector<ll>(block_cnt));
    //T1 前 i 个块 -> T2 第 j 个块
    vector<ll> val(n), left(block_cnt), right(block_cnt);

    for(int i = 0; i < n; ++ i){
        ++ cnt[i / B][to[i] / B];
    }
    for(int i = 1; i < block_cnt; ++ i){
        for(int j = 0; j < block_cnt; ++ j){
            cnt[i][j] += cnt[i-1][j];
        }
    }

    auto brute_add = [&](int l, int r, ll w){
        for(int i = l; i <= r; ++ i){
            right[to[i] / B] += w;
            val[i] += w;
        }
    };
    auto brute_query = [&](int l, int r){
        ll ans = 0;
        for(int i = l; i <= r; ++ i){
            ans += val[fr[i]] + left[fr[i] / B];
        }
        return ans;
    };
    auto block_add = [&](int l, int r, ll w){
        if(l <= r){
            for(int i = 0; i < block_cnt; ++ i){
                right[i] += (cnt[r][i] - cnt[l-1][i]) * w;
            }
            for(int i = l; i <= r; ++ i){
                left[i] += w;
            }
        }
    };
    auto block_query = [&](int l, int r){
        ll ans = 0;
        for(int i = l; i <= r; ++ i){
            ans += right[i];
        }
        return ans;
    };
    auto add = [&](int l, int r, ll w){
        const int bl = l / B, br = r / B;
        if(bl == br){
            brute_add(l, r, w);
        } else {
            brute_add(l, R[bl], w);
            brute_add(L[br], r, w);
            block_add(bl + 1, br - 1, w);
        }
    };
    auto query = [&](int l, int r){
        const int bl = l / B, br = r / B;
        if(bl == br){
            return brute_query(l, r);
        } else {
            return brute_query(l, R[bl]) + 
                   brute_query(L[br], r) + 
                   block_query(bl + 1, br - 1);
        }
    };

    for(int i = 0; i < n; ++ i){
        add(tr1.dfn[i], tr1.dfn[i], value[i]);
    }
    vector<pair<int, int> > path;
    for(int i = 0; i < q; ++ i){
        int u, v;
        ll k;
        cin >> u >> v >> k;
        -- u;
        -- v;
        tr1.decomp(u, v, path);
        for(auto i : path){
            add(i.first, i.second, k);
        }
        tr2.decomp(u, v, path);
        ll ans = 0;
        for(auto i : path){
            ans += query(i.first, i.second);
        }
        cout << ans << '\n';
    }
    return 0;
}
