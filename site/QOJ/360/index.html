    JOISC2017 1A Cultivation

    Problem Link: https://qoj.ac/problem/360。

    题意

    给你 \(R*C\) 的矩阵上 \(n\) 个点，初始只有这 \(n\) 个点为黑。每次操作，你可以选择一个方向（上/下/左/右），并且对于每个黑点，染黑它对应方向的那个点（如果超界不管）。问最少多少次操作可以使得所有点均染黑。 \(1\leq R,C\leq 10^9,1\leq n\leq 300\)。

    转化题意：每个点最后染黑的点是一个矩形。设分别执行了 \(u,d,l,r\) 次上下左右操作，那么点 \((x,y)\) 染黑的点就是 \((x-u,y-l)\) 到 \((x+d,y+r)\) 这个矩形。一组 \((u,d,l,r)\) 合法就相当于这些矩形的并覆盖整个矩阵。最后求的是所有合法的 \((u,d,l,r)\) 中 \(u+d+l+r\) 最小多少。

    直接枚举，枚举一维都是不能接受的。考虑 \(l\) 能有多少中取值。如果一组 \((u,d,l,r)\) 合法，\((u,d,l-1,r+1)\) 也合法，那么我们不妨就不考虑 \((u,d,l,r)\)。反之，如果 \((u,d,l-1,r+1)\) 不合法，那么一定是最左边一列的某一个位置不能被覆盖了，所以此时的 \(l\in\{y_i-1|i\in[1,n]\}\)，一共只有 \(n\) 种可能。

    再考虑此时的 \(r\)。什么时候 \((u,d,l,r)\) 合法，但是 \((u,d,l,r-1)\) 不合法呢？此时不一定是最右边的位置没被覆盖了；但是必然存在 \(y_i+r=y_j-l-1\) 或者 \(y_i+r-m\)。所以此时的 \(r\in\{y_i-y_j-1|i,j\in[1,n],y_i>y_j\}\cup\{m-y_i|i\in[1,n]\}\)，一共 \(n^2\) 种可能。

    最后考虑 \(u,d\) 的取值。如果确定了 \(l,r\) 的值，那么每一列就会分成若干段，最上面一段未染色的，最下面一段未染色的，中间若干段未染色的。此时计算 \(\min\{u+d\}\) 是简单的。\(u\) 有一个下界，即最上面未染色段最长长度；\(d\) 同理；\(u+d\) 的下界是中间若干段未染色的最长长度。于是维护 \(U,D,mx\) 表示上述三个量，那么 \(u+d\geq \max(U+D, mx)\)。

    我们并不需要对于每一列计算 \(U,D,mx\)，我们只需要提取出那些 \(y\in\{y_i-l-1|i\in[1,n]\}\) 的列以及第 \(m\) 列即可。如果暴力枚举 \(l,r\) 最后计算 \(U,D,mx\) 的话，复杂度是 \(O(n^4)\) 的。

    考虑优化：从大到小扫 \(r\)，那么每一段未染色段会逐渐合并，可以用链表维护这个合并过程中 \(U,D,mx\) 的对应变化，这样的话总复杂度变为 \(O(n^3)\)，可以通过。

    实现细节比较多，一个好的实现方法是先离散化，然后提取出关键列建链表，倒序遍历可能的 \(r\) 后考虑每个位置到底会不会由染色变为未染色。

    //qoj360
#include <bits/stdc++.h>
using namespace std;

const int N = 310;
int n, m, K, le[N];
pair<int, int> a[N];
int px[N], py[N];
int nn, mm;
typedef long long ll;

vector<pair<int, int> > ed;// 在 first 时间， px[a[second].first] 行的 m 列增加了一次覆盖
vector<pair<int, pair<int, int> > > can;//在 first-L 时间，px[a[second.first].first] 行的 py[second.second]-1-L 列增加了一次覆盖

ll U, D, mx;
struct mylist{
    int pre[N], nxt[N];
    void del(int x){
        if(pre[x] == 0 && nxt[x] == nn + 1){
            mx = 2e9;
        } else if(pre[x] == 0){
            pre[nxt[x]] = pre[x];
            U = max(U, (ll)px[nxt[x]]);
        } else if(nxt[x] == nn + 1){
            nxt[pre[x]] = nxt[x];
            D = min(D, (ll)px[pre[x]]);
        } else {
            nxt[pre[x]] = nxt[x];
            pre[nxt[x]] = pre[x];
            mx = max(mx, (ll)px[nxt[x]] - px[pre[x]] - 1);
        }
    }
} lst[N];

ll calc(int L){//全体左移 L 位后的情况
    U = 1, D = n, mx = 0;
    ll ans = 1e18;
    U = px[1], D = px[nn];
    for(int i = 1; i <= mm; ++ i){//建出第 py[i]-L-1 列的链表
        if(py[i] - L > 1){
            for(int j = 1; j <= nn; ++ j){
                lst[i].pre[j] = j - 1;
                lst[i].nxt[j] = j + 1;
            }
            for(int j = 1; j <= nn; ++ j){//删除初始为空的位置
                if(le[j] - L > py[i] - L - 1){
                    lst[i].del(j);
                }
            }
        }
    }
    for(int j = 1; j <= nn; ++ j){//第 m 列的列表，初始都是满的
        mx = max(mx, (ll)px[j] - px[j-1] - 1);
        lst[0].pre[j] = j - 1;
        lst[0].nxt[j] = j + 1;
    }
    int xx = can.size() - 1, yy = ed.size() - 1;
    while(xx >= 0 || yy >= 0){//倒序枚举时间，删点
        if(yy < 0 || (xx >= 0 && can[xx].first - L >= ed[yy].first)){
            int x = can[xx].second.first, y = can[xx].second.second;
            if(can[xx].first - L > 0){
                ans = min(ans, can[xx].first - L + max(U - 1 + n - D, mx));
                //考虑链表 y 是否要删除元素 x，即 x 是该行最后的元素 或者 x 同行下一个元素 - L > 目前这一列
                if(px[a[x+1].first] != px[a[x].first] || py[a[x+1].second] - L > py[y] - L - 1){
                    lst[y].del(a[x].first);
                }
            }
            -- xx;
        } else {
            int x = ed[yy].second;
            if(ed[yy].first > 0){
                ans = min(ans, ed[yy].first + max(U - 1 + n - D, mx));
                if(px[a[x+1].first] != px[a[x].first]){
                    lst[0].del(a[x].first);
                }
            }
            -- yy;
        }
    }
    //该删的都删了，如果还可以的话 R 就可以 =0。
    ans = min(ans, max(U - 1 + n - D, mx));
    return ans;
}

int main(){
    scanf("%d%d%d", &n, &m, &K);
    for(int i = 1; i <= K; ++ i){
        scanf("%d%d", &a[i].first, &a[i].second);
        px[i] = a[i].first;
        py[i] = a[i].second;
        le[i] = 1e9;
    }
    sort(a + 1, a + K + 1);
    sort(px + 1, px + K + 1);
    sort(py + 1, py + K + 1);
    nn = unique(px + 1, px + K + 1) - px - 1;
    mm = unique(py + 1, py + K + 1) - py - 1;
    for(int i = 1; i <= K; ++ i){
        ed.push_back(make_pair(m - a[i].second, i));
        a[i].first = lower_bound(px + 1, px + nn + 1, a[i].first) - px;
        le[a[i].first] = min(le[a[i].first], a[i].second);
        a[i].second = lower_bound(py + 1, py + mm + 1, a[i].second) - py;
    }
    for(int i = 1; i <= mm; ++ i){
        for(int j = 1; j <= K; ++ j){
            if(py[i] > py[a[j].second]){
                can.push_back(make_pair(py[i] - py[a[j].second] - 1, make_pair(j, i)));
            }
        }
    }
    sort(ed.begin(), ed.end());
    sort(can.begin(), can.end());
    ll ans = 1e18;
    for(int i = 1; i <= K; ++ i){
        ans = min(ans, py[a[i].second] - 1 + calc(py[a[i].second] - 1));
    }
    printf("%lld\n", ans);
    return 0;
}
