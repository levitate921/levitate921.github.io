    P9394 白鹭兰

    Problem Link: https://www.luogu.com.cn/problem/P9394。

    题意

    给定一张 \(n\) 个点，\(m\) 条边的无向图 \(G\)，求一个最小的 \(k\)，使得存在点集划分 \(V_1,V_2,...,V_t\) 满足：对于每一个点集的前缀/后缀并，它们的导出子图联通，且每个点集大小 \(\leq k\)。求最小的 \(k\) 和划分方案。 \(2\leq n \leq 2\times 10^5,1\leq m\leq 2.3\times 10^5\)，保证无重边自环，保证图联通。

    前置知识是双极定向。

    双极定向，即给定一个无向图 \(G\) 以及两个点 \(S,T\)，要求你找到一种给这个图边定向的方式，使得：

    • 这张图是一个 DAG；
    • \(S\) 是唯一一个入度为 \(0\) 的点，\(T\) 是唯一一个出度为 \(0\) 的点。

    有个等价的问题，就是转化为求一个拓扑序，满足：每个前缀联通，每个后缀联通，然后每条边从拓扑序小的连向拓扑序大的就行。

    结论是：一张图可以被双极定向，当且仅当这张图建圆方树后方点构成一条链，且所有方点都在 \(S\to T\) 的路径上，形象地描述，就是以 \(S,T\) 为两端的毛毛虫。

    一个求解双极定向的方法1是：

    首先将这个圆方树拆成若干个点双，每个点双内部有两个点，为割点或者是 \(S,T\)，把这两个点当作这个点双内部的 \(S,T\)，就只用考虑对于一个点双求双极定向。

    每个点维护一个列表，表示这个点放入拓扑序后，这个点列表内的点都能放到拓扑序中。

    • 以 \(S\) 为根跑 tarjan，求出每个点的 \(fa,low\)。
    • 然后开始每次剥掉一个叶子：按照 dfn 序倒序枚举点 \(x\)，如果 \(x\) 不在 \(S\to T\) 的路径上，将点 \(x\) 插入到 \(fa_x,low_x\) 对应的列表中，本质是缩二度点。
    • 最后形成一条链，从 \(S\) 到 \(T\) 依次将点放到拓扑序中，然后递归其列表中的点即可。注意一个点只插入一次。

    对于原题，可以发现双极定向的等价问题和原题很像，那么就是，如果给出的图可以双极定向，那么答案 \(k=1\)，输出一个双极定向的拓扑序方案即可。否则，我们需要在给出的图圆方树上将若干连通块缩成若干点，使得缩点后的圆方树可以双极定向，且最大的缩成一个点的连通块最小。

    如果枚举 \(S,T\) 是好做的，我们考虑能否 dp 求出最优的 \(S,T\)。设 \(f_i\) 表示 \(S\) 在 \(i\) 子树外，\(T\) 在子树内，子树内缩点后最大连通块最小是多少。

    定义一个点的 \(siz\)：

    • 如果这个点是圆点，\(siz\) 为子树内圆点个数。
    • 如果这个点是方点，\(siz\) 为它的儿子圆点的 \(siz\) 最大值。

    容易发现，\(f_i\) 可以从任意一个 \(siz_j\) 在儿子内最大的 \(j\) 转移而来：

    • 如果最大的 \(siz_j\) 唯一且不从这个点转移，那么 \(f_i=siz_j\)；如果从这个点转移才有可能 \(<siz_j\):
    • 如果最大的 \(siz_j\) 不唯一，那么 \(f_i=siz_j\) 无法改变了，从哪转移都可以了。

    最后是枚举 lca \(x\) 以及它的两个子树 \(y,z\) 统计答案，容易发现 \(siz_y,siz_z\) 也取 \(x\) 儿子内最大的两个子树就行。

    我们根据 dp 的转移可以找到 \(S,T\)，最后缩点后跑双极定向即可。复杂度 \(O(n+m)\)。

    // Problem: P9394 白鹭兰
// Contest: Luogu
// URL: https://www.luogu.com.cn/problem/P9394
// Memory Limit: 512 MB
// Time Limit: 1000 ms
// 
// Powered by CP Editor (https://cpeditor.org)

#include <bits/stdc++.h>
using namespace std;

const int N = 5e5 + 10;
int n, m;
vector<int> g[N], tr[N];

int dfn[N], low[N], st[N], ins[N], tp, dfc;
int cnt;
void tarjan(int x){
    dfn[x] = low[x] = ++ dfc;
    st[++tp] = x;
    ins[x] = 1;
    for(int i : g[x]){
        if(!dfn[i]){
            tarjan(i);
            low[x] = min(low[x], low[i]);
            if(low[i] >= dfn[x]){
                ++ cnt;
                while(true){
                    tr[cnt+n].push_back(st[tp]);
                    tr[st[tp]].push_back(cnt+n);
                    ins[st[tp]] = 0;
                    -- tp;
                    if(st[tp+1] == i){
                        break;
                    }
                }
                tr[cnt+n].push_back(x);
                tr[x].push_back(cnt+n);
            }
        } else {
            low[x] = min(low[x], dfn[i]);
        }
    }
}

int siz[N], f[N], fr[N], dep[N], anc[N];
int ans, RT, xs, ys;
void dfs(int x, int fa){
    dep[x] = dep[fa] + 1;
    anc[x] = fa;
    siz[x] = (x <= n ? 1 : 0);
    pair<int, int> tmp = make_pair(0, 0);
    vector<pair<int, int> > v;
    for(int i : tr[x]){
        if(i == fa){
            continue;
        }
        dfs(i, x);
        siz[x] += siz[i];
        tmp = max(tmp, make_pair(siz[i], i));
        v.push_back(make_pair(siz[i], i));
    }
    if(x <= n){
        f[x] = 1;
    }
    int pos = tmp.second;
    fr[x] = pos;
    for(int i : tr[x]){
        if(i == fa){
            continue;
        }
        if(i != pos){
            if(x <= n){
                f[x] += siz[i];
            } else {
                f[x] = max(f[x], siz[i]);
            }
        }
    }
    sort(v.begin(), v.end());
    reverse(v.begin(), v.end());
    if(x <= n){
        if(v.size() > 1){
            int val = max({f[x]+n-siz[x]-siz[v[1].second],f[v[0].second],f[v[1].second]});
            if(val < ans){
                ans = val;
                RT = x;
                xs = v[0].second;
                ys = v[1].second;
            }
        }
    } else {
        if(v.size() > 1){
            if(v.size() == 2){
                int val = max({f[v[0].second], f[v[1].second], n-siz[x]});
                if(val < ans){
                    ans = val;
                    RT = x;
                    xs = v[0].second;
                    ys = v[1].second;
                }
            } else {
                int val = max({f[v[0].second], f[v[1].second], siz[v[2].second], n-siz[x]});
                if(val < ans){
                    ans = val;
                    RT = x;
                    xs = v[0].second;
                    ys = v[1].second;
                }
            }
        }
    }
    f[x] = max(f[x], f[pos]);
}

int inr[N], id[N], mn[N], mx[N];
vector<int> bel[N];
void mg(int x, int fa, int rt){
    if(x <= n){
        bel[rt].push_back(x);
    }
    for(int i : tr[x]){
        if(i == fa || inr[i]){
            continue;
        }
        mg(i, x, rt);
    }
}

vector<pair<int, int> > eg[N];
int sta[N], ed[N];
bool cmp(int x, int y){
    return mn[x] < mn[y];
}
void pr(int x){
    if(!bel[x].size()){
        return;
    }
    printf("%d ", bel[x].size());
    for(int i : bel[x]){
        printf("%d ", i);
    }
    vector<int> ().swap(bel[x]);
    puts("");
}

vector<int> G[N], P[N];
int Dfn[N], Low[N], Fa[N], idx, Que[N], On[N], Vis[N];
void tar(int x, int fa){
    Fa[x] = fa;
    Dfn[x] = Low[x] = ++ idx;
    Que[idx] = x;
    for(int i : G[x]){
        if(!Dfn[i]){
            tar(i, x);
            Low[x] = min(Low[x], Low[i]);
        } else {
            Low[x] = min(Low[x], Dfn[i]);
        }
    }
}
void dfs__(int x){
    Vis[x] = 1;
    pr(x);
    for(int i : P[x]){
        if(!Vis[i]){
            dfs__(i);
        }
    }
}

void solve(int id){
    if(eg[id].size() == 1){
        pr(sta[id]);
        pr(ed[id]);
        return;
    }
    for(auto i : eg[id]){
        G[i.first].push_back(i.second);
        G[i.second].push_back(i.first);
    }
    tar(sta[id], 0);
    vector<int> path;
    for(int x = ed[id]; x; x = Fa[x]){
        On[x] = 1;
        path.push_back(x);
    }
    reverse(path.begin(), path.end());
    for(int i = idx; i >= 1; -- i){
        int x = Que[i];
        if(!On[x]){
            P[Fa[x]].push_back(x);
            P[Que[Low[x]]].push_back(x);
        }
    }
    for(int x : path){
        dfs__(x);
    }
    for(int i = 1; i <= idx; ++ i){
        int x = Que[i];
        Dfn[i] = Low[i] = Fa[i] = On[i] = Vis[i] = 0;
        vector<int> ().swap(G[i]);
        vector<int> ().swap(P[i]);
    }
    idx = 0;
}

int main(){
    scanf("%d%d", &n, &m);
    ans = n;
    for(int i = 1; i <= m; ++ i){
        int u, v;
        scanf("%d%d", &u, &v);
        g[u].push_back(v);
        g[v].push_back(u);
    }
    tarjan(1);
    int rt = 1;
    while(tr[rt].size() == 1){
        ++ rt;
    }
    dfs(rt, 0);
    int S = xs, T = ys;
    id[RT] = n + n;
    id[S] = n + n - 1;
    id[T] = n + n + 1;
    inr[RT] = inr[S] = inr[T] = 1;
    while(tr[S].size() != 1){
        id[fr[S]] = id[S] - 1;
        S = fr[S];
        inr[S] = 1;
    }
    while(tr[T].size() != 1){
        id[fr[T]] = id[T] + 1;
        T = fr[T];
        inr[T] = 1;
    }
    int rc = 0;
    for(int i = 1; i <= n + cnt; ++ i){
        mn[i] = 1e9;
        if(inr[i]){
            if(i <= n){
                mg(i, 0, i);
                ++ rc;
            } else {
                for(auto j : tr[i]){
                    if(!inr[j]){
                        inr[j] = 1;
                        mg(j, 0, j);
                        ++ rc;
                    }
                }
            }
        }
    }
    printf("%d %d\n", ans, rc);
    for(int i = 1; i <= n; ++ i){
        for(auto j : g[i]){
            if(inr[i] && inr[j] && (dep[i] < dep[j] || (dep[i] == dep[j] && i < j))){
                int bel = anc[j] - n;
                if(id[i]){
                    mn[bel] = min(mn[bel], id[i]);
                    mx[bel] = max(mx[bel], id[i]);
                    if(id[i] == mn[bel]){
                        sta[bel] = i;
                    }
                    if(id[i] == mx[bel]){
                        ed[bel] = i;
                    }
                }
                if(id[j]){
                    mn[bel] = min(mn[bel], id[j]);
                    mx[bel] = max(mx[bel], id[j]);
                    if(id[j] == mn[bel]){
                        sta[bel] = j;
                    }
                    if(id[j] == mx[bel]){
                        ed[bel] = j;
                    }
                }
                eg[bel].emplace_back(i, j);
            }
        }
    }
    static int shunxu[N];
    for(int i = 1; i <= cnt; ++ i){
        shunxu[i] = i;
    }
    sort(shunxu + 1, shunxu + cnt + 1, cmp);
    for(int i = 1; i <= cnt; ++ i){
        if(!sta[shunxu[i]]){
            break;
        }
        solve(shunxu[i]);
    }
    return 0;
}
