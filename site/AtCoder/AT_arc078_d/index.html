    ARC078F Mole and Abandoned Mine

    Problem Link: https://www.luogu.com.cn/problem/AT_arc078_d。

    题意

    有一张简单连通无向图，你需要删除若干条边使得删除后的图 \(1\to n\) 只有一条简单路径，删除一条边的代价为边权。求满足条件的最小代价。 \(n\leq 15\)，可以做到 \(n\leq 17\)。

    发现，合法的删边后的图形如，一条 \(1\to n\) 的链，链上每个点可以挂一个连通块。于是想到状压 dp，设 \(f_{i,S}\) 表示目前路径 \(1\to i\)，路径上的点和挂在路径上的点的点集合为 \(S\)。转移枚举一个 \(j,j\in T\)，预处理将连通块 \(T\) 单独分割出来的代价，然后如果 \(i\to j\) 这条边存在，就可以转移到 \(f_{j,S^T}\)，复杂度 \(O(3^nn^2)\)，可以通过原题。

    但是还能更优：注意到转移形如 \(f_{i,S} - val_{i\to j} + g_{T} \to f_{j,T}\)，而 \(f_{i,S} - val_{i\to j}\) 一项和 \(T\) 无关，预处理这个的最优值为 \(h_{j,S}\) 即可。复杂度 \(O(3^nn)\)。

    似乎有 \(O(3^n)\) 做法？但我不会。

    //AT_arc078_d
#include <bits/stdc++.h>
using namespace std;

int main(){
    int n, m;
    cin >> n >> m;
    vector<vector<int> > g(n, vector<int>(n));
    int sum = 0;
    while(m --){
        int u, v, w;
        cin >> u >> v >> w;
        -- u;
        -- v;
        sum += w;
        g[u][v] = g[v][u] = w;
    }
    vector<vector<int> > F(n, vector<int>(1<<n));
    vector<vector<int> > tmp(n, vector<int>(1<<n));
    vector<int> G(1<<n);
    for(int S = 0; S < 1 << n; ++ S){
        for(int i = 0; i < n; ++ i){
            F[i][S] = -1e9;
            tmp[i][S] = -1e9;
        }
    }
    for(int S = 0; S < 1 << n; ++ S){
        for(int i = 0; i < n; ++ i){
            if(S & (1 << i)){
                for(int j = i + 1; j < n; ++ j){
                    if(S & (1 << j)){
                        G[S] += g[i][j];
                    }
                }
            }
        }
        if(S & 1){
            F[0][S] = G[S];
        }
    }

    for(int S = 0; S < 1 << n; ++ S){
        for(int i = 0; i < n; ++ i){
            if(!(S & (1 << i))){
                continue;
            }
            for(int j = 0; j < n; ++ j){
                if(g[i][j] && ((S ^ (1 << i)) & (1 << j))){
                    tmp[i][S^(1<<i)] = max(tmp[i][S^(1<<i)], F[j][S^(1<<i)] + g[i][j]);
                }
            }
            for(int T = S ^ (1 << i); T; T = (T - 1) & (S ^ (1 << i))){
                F[i][S] = max(F[i][S], G[S^T] + tmp[i][T]);
            }
        }
    }
    cout << sum - F[n-1][(1<<n)-1] << '\n';
    return 0;
}
